```cpp
#include <iostream>
#include <clocale>
using namespace std;

void han(int n, int start, int point, int temp) {
    if (n <= 0)
        return;

    han(n - 1, start, temp, point);
    moveDisk(start, point);
    han(n - 1, temp, point, start);
}

int main() {
    setlocale(LC_ALL, "ru");

    int n;
    cout << "Введите количество дисков: ";
    cin >> n;

    han(n, 1, 3, 2);

	return 0;
```

Реализация c визуализацией:

```cpp
#include <iostream>
#include <clocale>
using namespace std;

int N;
int tower1[10], h1 = 0;
int tower2[10], h2 = 0;
int tower3[10], h3 = 0;
int moveCount = 0;

void printTowers() {
    cout << "\n";
    for (int level = N - 1; level >= 0; level--) {
        // Стержень 1
        if (level < h1) {
            int disk = tower1[level];
            for (int s = 0; s < N - disk; s++) cout << " ";
            for (int d = 0; d < 2 * disk - 1; d++) cout << "=";
            for (int s = 0; s < N - disk; s++) cout << " ";
        }
        else {
            for (int s = 0; s < N - 1; s++) cout << " ";
            cout << "|";
            for (int s = 0; s < N - 1; s++) cout << " ";
        }
        cout << "  ";

        // Стержень 2
        if (level < h2) {
            int disk = tower2[level];
            for (int s = 0; s < N - disk; s++) cout << " ";
            for (int d = 0; d < 2 * disk - 1; d++) cout << "=";
            for (int s = 0; s < N - disk; s++) cout << " ";
        }
        else {
            for (int s = 0; s < N - 1; s++) cout << " ";
            cout << "|";
            for (int s = 0; s < N - 1; s++) cout << " ";
        }
        cout << "  ";

        // Стержень 3
        if (level < h3) {
            int disk = tower3[level];
            for (int s = 0; s < N - disk; s++) cout << " ";
            for (int d = 0; d < 2 * disk - 1; d++) cout << "=";
            for (int s = 0; s < N - disk; s++) cout << " ";
        }
        else {
            for (int s = 0; s < N - 1; s++) cout << " ";
            cout << "|";
            for (int s = 0; s < N - 1; s++) cout << " ";
        }
        cout << endl;
    }
    cout << "-----------------------------" << endl;
}

void moveDisk(int from, int to) {
    int disk;

    if (from == 1) {
        disk = tower1[--h1];
    }
    else if (from == 2) {
        disk = tower2[--h2];
    }
    else {
        disk = tower3[--h3];
    }

    if (to == 1) {
        tower1[h1++] = disk;
    }
    else if (to == 2) {
        tower2[h2++] = disk;
    }
    else {
        tower3[h3++] = disk;
    }

    moveCount++;
    cout << "Ход " << moveCount << ": " << from << " => " << to << endl;
    printTowers();
}

void han(int n, int start, int point, int temp) {
    if (n <= 0)
        return;

    han(n - 1, start, temp, point);
    moveDisk(start, point);
    han(n - 1, temp, point, start);
}

int main() {
    setlocale(LC_ALL, "ru");

    int n;
    cout << "Введите количество дисков: ";
    cin >> n;
    N = n;

    for (int i = 0; i < N; i++) {
        tower1[i] = N - i;
    }
    h1 = N;

    cout << "\nНачальное состояние:" << endl;
    printTowers();

    han(n, 1, 3, 2);

    cout << "Башня собрана! Всего ходов: " << moveCount << endl;

    return 0;
}
```
